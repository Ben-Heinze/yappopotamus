#+TITLE: Spotify Playlist Data
#+OPTIONS: toc:nil num:nil

* Spotify PLaylist Data 

The most common pattern: print output to stdout and capture it with ~:results output~.

#+NAME: python-example
#+begin_src python :results output
import spotipy
from spotipy.oauth2 import SpotifyOAuth

sp = spotipy.Spotify(auth_manager=SpotifyOAuth(
    client_id="YOUR_CLIENT_ID",
    client_secret="YOUR_CLIENT_SECRET",
    redirect_uri="http://localhost:8888/callback",
    scope="playlist-read-private"
))

playlist_id = "YOUR_PLAYLIST_ID"

results = sp.playlist_items(playlist_id)
tracks = results["items"]

# Paginate — the API returns max 100 items per request
while results["next"]:
    results = sp.next(results)
    tracks.extend(results["items"])

for item in tracks:
    track = item["track"]
    if track is None:  # can happen with removed/local tracks
        continue
    name = track["name"]
    duration_ms = track["duration_ms"]
    added_at = item["added_at"]  # e.g. "2024-03-15T22:41:07Z"
    minutes, seconds = divmod(duration_ms // 1000, 60)
    print(f"{name} — {minutes}:{seconds:02d} — added {added_at}")
#+end_src
