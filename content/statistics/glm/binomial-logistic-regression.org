#+TITLE: Binomial Logistic Regression
#+OPTIONS: toc:nil num:nil

* Binomial Logistic Regression

#+begin_src R :results output
set.seed(42)
n <- 50
trials <- 10
x <- rnorm(n)
log_odds <- -0.5 + 1.5 * x
p <- 1 / (1 + exp(-log_odds))
successes <- rbinom(n, size = trials, prob = p)

model <- glm(cbind(successes, trials - successes) ~ x, family = binomial(link = "logit"))
summary(model)
#+end_src
