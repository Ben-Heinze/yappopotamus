#+TITLE: Binary Logistic Regression
#+OPTIONS: toc:nil num:nil

* Binary Logistic Regression



#+begin_src R :results output
set.seed(42)
n <- 200
x <- rnorm(n)
log_odds <- -1 + 2 * x
p <- 1 / (1 + exp(-log_odds))
y <- rbinom(n, size = 1, prob = p)

model <- glm(y ~ x, family = binomial(link = "logit"))
summary(model)
#+end_src
